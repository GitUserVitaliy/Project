#include <iostream>
#include <string>

using namespace std;

int get_symbol(int symbol)
{
	symbol = symbol % 26 + 65;
	return symbol;
}

int main()
{
	int var = 0;
	cout << "Enter number: ";
	cin >> var;
	char s = get_symbol(var);
	cout << "Symbol: " << s << endl;


	system("pause");
	return 0;
}
