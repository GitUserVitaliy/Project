string GetHash(string word)
{
	string hash;
	char symbol;
	const int key = 1;
	
	for (int i = 0; i < word.length(); i++)
	{
		symbol = (word[i] * (word.length() - i) * key);
		symbol = 64 + symbol % 26;
		
		hash.push_back(symbol);
		
	}




	return hash;
}
