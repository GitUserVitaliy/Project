char GetSymbol(int symbol)
{
	char csymbol;
	csymbol = symbol % 26 + 65;
	return csymbol;
}

int GetNumber(char symbol)
{
	int number = symbol;
	return number;
}

char GetVar(int symbol)
{
	char csymbol;
	csymbol = symbol % 10 + 48;
	return csymbol;
}

string GetHash(string word)
{
	if (word.length() > 16)
	{
		return "@error@";
	}
	else
	{
		string hash;
		string salt = "safe";
		int size = word.length();
		int rand_var = 0;
		int Vsize = 16;
		vector<int> arr;
		int num[20] = {13,23,54,756,2,34,12,87,87,456,34,23,76,89,23,54,23,54,65,32};
		for (int i = 0; i < Vsize; i++)
		{
			if (i == size)
			{
				Vsize = Vsize - i;
				i = 0;
			}
			arr.push_back((int)word[i] * num[i] * (Vsize - i)*size);
		}
		for (int j = 0; j < salt.length(); j++)
		{
			arr.push_back((int)salt[j] * num[j] * (salt.length() - j) *size);
		}
		
		for (int i = 0; i < arr.size(); i++)
		{
			if (arr[i] % 7 == 0)
			{
				hash.push_back(GetVar(arr[i] / 7));
			}
			else
			{
				hash.push_back(GetSymbol(arr[i]));
			}
		}
		return hash;
	}
}
